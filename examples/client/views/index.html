// ── Connect to the SSE stream ──────────────────────────
//
// Query params:
//   userId  – identifies this user across all their tabs
//   room    – auto-joins a group on connect
//   topics  – comma-separated topics for sticky replay

const createSSE = (userId, room, topics) => {
  const params = new URLSearchParams({
    userId,
    room,
    topics: topics.join(','),
  });

  const es = new EventSource(
    `http://localhost:3000/events?${params}`
  );

  // ── Connected ─────────────────────────────────────────
  es.addEventListener('connected', (e) => {
    const { clientId } = JSON.parse(e.data);
    console.log('✅ Connected as', clientId);
  });

  // ── Generic message handler ────────────────────────────
  es.onmessage = (e) => {
    console.log('message:', e.data);
  };

  // ── Connection error ───────────────────────────────────
  es.onerror = () => {
    console.warn('SSE connection lost, browser will retry…');
  };

  return es;
};

const es = createSSE('alice', 'team-backend', ['status', 'orders']);

OnConnect: func(ctx *fiber.Ctx, client *sse.Client) error {
  room := ctx.Query("room")
  if room != "" {
    hub.JoinGroup(client.ID, room)
  }
  return nil
},

// ── Auto-reconnect wrapper with backoff ────────────────
// The browser retries SSE automatically, but this wrapper
// gives you lifecycle hooks and exponential backoff control.

class SSEClient {
  constructor(url, options = {}) {
    this.url        = url;
    this.options    = options;
    this.handlers   = {};
    this.retryDelay = 1000;
    this.maxDelay   = 30000;
    this._connect();
  }

  _connect() {
    this.es = new EventSource(this.url);

    this.es.onopen = () => {
      this.retryDelay = 1000;  // reset on success
      this.options.onConnect?.();
    };

    this.es.onerror = () => {
      this.es.close();
      this.options.onDisconnect?.();
      setTimeout(() => this._connect(), this.retryDelay);
      this.retryDelay = Math.min(this.retryDelay * 2, this.maxDelay);
    };

    // Re-attach all registered handlers
    Object.entries(this.handlers).forEach(([type, fn]) => {
      this.es.addEventListener(type, fn);
    });
  }

  on(type, fn) {
    this.handlers[type] = fn;
    this.es.addEventListener(type, fn);
    return this;
  }

  close() { this.es.close(); }
}

// Usage
const client = new SSEClient(
  'http://localhost:3000/events?userId=alice',
  {
    onConnect:    () => console.log('connected'),
    onDisconnect: () => console.log('reconnecting…'),
  }
)
  .on('connected',    (e) => console.log('welcome', JSON.parse(e.data)))

  .on('notification',(e) => showBanner(e.data))
  .on('deploy',      (e) => refreshDashboard(e.data));
